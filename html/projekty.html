        import pandas as pd
        import numpy as np
        import matplotlib.pyplot as plt
        import folium
        import geopy
        from sklearn import preprocessing, cluster
        import scipy
        import seaborn as sns
        
        # Načítanie dát
        dtf = pd.read_csv("directory.csv")
        dtf = dtf[dtf["City"]=="Dubai"]
        dtf = dtf[["City","Street Address","Longitude","Latitude"]]
        
        # Generovanie náhodných dát pre príklad
        dtf["Potential"] = np.random.randint(low=3, high=11, size=len(dtf))
        dtf["Staff"] = dtf["Potential"].apply(lambda x: int(np.random.rand()*x)+1)
        dtf["Capacity"] = dtf["Potential"] - dtf["Staff"]
        dtf["Cost"] = np.random.choice(["high","medium","low"], size=len(dtf), p=[0.4,0.5,0.1])
        
        # Získanie GPS súradníc pre mesto
        city = "Dubai"
        locator = geopy.geocoders.Nominatim(user_agent="MyCoder")
        location = locator.geocode(city)
        location = [location.latitude, location.longitude]
        
        # Priradenie farieb podľa nákladov
        colors = []
        for i in dtf["Cost"]:
          if(i == "high"):
            colors.append("red")
          if(i == "medium"):
            colors.append("orange")
          if(i == "low"):
            colors.append("green")
        dtf["color"] = colors
        
        # Inicializácia mapy
        maps = folium.Map(location=location, tiles="cartodb positron")
        
        # Pridanie bodov na mapu
        dtf.apply(lambda row: folium.CircleMarker(
                   location=[row["Latitude"],row["Longitude"]], popup=row["Street Address"],
                   color=row["color"], fill=True,
                   radius=row["Staff"]).add_to(maps), axis=1)

            int main() {
                // Počet obdĺžnikov na vstupe
                int number_of_rectangles;
                cin >> number_of_rectangles;
            
                // Načítanie výšok obdĺžnikov do vektora
                vector length_rectangle(number_of_rectangles);
                for (int i = 0; i < number_of_rectangles; ++i) {
                    cin >> length_rectangle[i];
                }
            
                // Premenná pre maximálnu plochu
                int max_area = 0;
            
                // Iterácia cez všetky obdĺžniky
                for (int i = 0; i < number_of_rectangles; ++i) {
                    int min_height = length_rectangle[i]; // Aktuálna minimálna výška
                    // Ak je teoretická plocha menšia než max_area, pokračujeme
                    if (min_height * (number_of_rectangles - i) <= max_area) {
                        continue;
                    }
            
                    // Iterácia cez nasledujúce obdĺžniky
                    for (int j = i; j < number_of_rectangles; ++j) {
                        if (min_height > length_rectangle[j]) {
                            min_height = length_rectangle[j]; // Aktualizujeme minimálnu výšku
                        }
                        if (min_height == 0) {
                            break; // Ak je výška 0, skončíme
                        }
                        int area = min_height * (j - i + 1); // Výpočet plochy
                        if (area > max_area) {
                            max_area = area; // Aktualizácia maximálnej plochy
                        }
                    }
                }
            
                cout << max_area << endl; // Výstup maximálnej plochy
            
                return 0;
            }
